# 路由导航

## TanStack Router

项目使用 `@tanstack/react-router` 进行路由管理，采用文件路由模式。

### 路由文件结构

```
src/routes/
├── __root.tsx           # 根路由：ThemeProvider + TooltipProvider + Toaster
├── index.tsx            # / 重定向到 /dashboard
├── dashboard.tsx        # /dashboard
├── books.tsx            # /books (含子路由)
├── books.index.tsx      # /books/ (账本列表)
├── books.$bookId.tsx    # /books/:bookId (账本详情)
├── chatbot.tsx          # /chatbot
├── customers.tsx        # /customers
├── customers.index.tsx  # /customers/
├── products.tsx         # /products
├── products.index.tsx   # /products/
├── orders.tsx           # /orders
├── orders.index.tsx     # /orders/
└── settings.tsx         # /settings
```

### 路由定义模板

简单路由（无子路由）：

```typescript
// src/routes/chatbot.tsx
import { createFileRoute } from '@tanstack/react-router'
import { ChatbotPage } from '@/pages/chatbot/chatbot-page'
import { AppLayout } from '@/components/layouts/app-layout'

export const Route = createFileRoute('/chatbot')({
  component: () => (
    <AppLayout>
      <ChatbotPage />
    </AppLayout>
  ),
})
```

带子路由的路由：

```typescript
// src/routes/books.tsx
import { createFileRoute, Outlet } from '@tanstack/react-router'
import { AppLayout } from '@/components/layouts/app-layout'

export const Route = createFileRoute('/books')({
  component: () => (
    <AppLayout>
      <Outlet />
    </AppLayout>
  ),
})
```

子路由指向页面组件：

```typescript
// src/routes/books.$bookId.tsx
import { createFileRoute } from '@tanstack/react-router'
import { BookDetailPage } from '@/pages/books/book-detail-page'

export const Route = createFileRoute('/books/$bookId')({
  component: BookDetailPage,
})
```

### 根路由

```typescript
// src/routes/__root.tsx
import { createRootRoute, Outlet } from '@tanstack/react-router'
import { ThemeProvider } from '@/components/theme-provider'
import { TooltipProvider } from '@/components/ui/tooltip'
import { Toaster } from '@/components/ui/sonner'

export const Route = createRootRoute({
  component: () => (
    <ThemeProvider defaultTheme="system" storageKey="theme">
      <TooltipProvider>
        <Outlet />
        <Toaster />
      </TooltipProvider>
    </ThemeProvider>
  ),
})
```

## 导航方式

### 编程式导航

```typescript
import { useRouter } from '@tanstack/react-router'

const router = useRouter()
const navigate = () => router.navigate({ to: '/dashboard' })
```

### 声明式导航

```typescript
import { Link } from '@tanstack/react-router'

<Link to="/chatbot">聊天</Link>
```

## 路由树自动生成

路由树通过 `routeTree.gen.ts` 自动生成，无需手动维护。添加新的路由文件后运行开发服务器即可自动更新。
